4) Native wheel listener with { passive: false } so preventDefault actually works:

useEffect(() => {
  const el = wheelContainerRef.current;
  if (!el || !lightboxPhotoId) return;
  const onWheel = (e: WheelEvent) => {
    e.preventDefault();
    const factor = e.deltaY < 0 ? 1.15 : 1 / 1.15;
    setZoom(z => Math.max(0.25, Math.min(5, z * factor)));
  };
  el.addEventListener('wheel', onWheel, { passive: false });
  return () => el.removeEventListener('wheel', onWheel);
}, [lightboxPhotoId]);

5) Keyboard +/-/0 — extend the existing Esc keyboard useEffect (or add a sibling). Skip when editingNote is set (the existing ref pattern at line 45 should be reused so stale closure is avoided):

useEffect(() => {
  if (!lightboxPhotoId) return;
  const onKey = (e: KeyboardEvent) => {
    // existing editingNote ref check pattern — DON'T intercept while editing a note
    if (editingNoteRef.current) return;
    if (e.key === '+' || e.key === '=') { e.preventDefault(); setZoom(z => Math.min(5, z * 1.25)); }
    else if (e.key === '-' || e.key === '_') { e.preventDefault(); setZoom(z => Math.max(0.25, z / 1.25)); }
    else if (e.key === '0') { e.preventDefault(); setZoom(1); setPan({ x: 0, y: 0 }); }
  };
  window.addEventListener('keydown', onKey);
  return () => window.removeEventListener('keydown', onKey);
}, [lightboxPhotoId]);

6) Mouse drag-pan + double-click toggle handlers (regular React handlers — no preventDefault concerns since they're not passive):

const onMouseDown = (e: React.MouseEvent) => {
  if (zoom <= 1) return;
  e.preventDefault();
  dragStartRef.current = { x: e.clientX, y: e.clientY, panX: pan.x, panY: pan.y };
};
const onMouseMove = (e: React.MouseEvent) => {
  const ds = dragStartRef.current;
  if (!ds) return;
  setPan({ x: ds.panX + (e.clientX - ds.x), y: ds.panY + (e.clientY - ds.y) });
};
const onMouseUp = () => { dragStartRef.current = null; };
const onDoubleClick = (e: React.MouseEvent) => {
  e.stopPropagation();
  if (zoom === 1) setZoom(2);
  else { setZoom(1); setPan({ x: 0, y: 0 }); }
};
