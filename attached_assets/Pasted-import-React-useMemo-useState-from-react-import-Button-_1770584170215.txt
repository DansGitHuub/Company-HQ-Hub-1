function roundTo(n: number, decimals = 2) {
  const p = Math.pow(10, decimals);
  return Math.round(n * p) / p;
}

function clamp(n: number, min?: number, max?: number) {
  if (min !== undefined && n < min) return min;
  if (max !== undefined && n > max) return max;
  return n;
}

/** --- Unit conversions (MVP set; extend freely) --- */
function ftToIn(ft: number) { return ft * 12; }
function inToFt(inches: number) { return inches / 12; }
function sqFtToAcres(sqft: number) { return sqft / 43560; }
function acresToSqFt(acres: number) { return acres * 43560; }
function cuFtToCuYd(cuft: number) { return cuft / 27; }
function cuYdToCuFt(cuyd: number) { return cuyd * 27; }
function gallonsToLiters(gal: number) { return gal * 3.78541; }
function litersToGallons(l: number) { return l / 3.78541; }
function lbsToKg(lbs: number) { return lbs * 0.45359237; }
function kgToLbs(kg: number) { return kg / 0.45359237; }
function tonsToLbs(tons: number) { return tons * 2000; }
function lbsToTons(lbs: number) { return lbs / 2000; }
function cmToIn(cm: number) { return cm / 2.54; }
function inToCm(inches: number) { return inches * 2.54; }
function mToFt(m: number) { return m * 3.28084; }
function ftToM(ft: number) { return ft / 3.28084; }

function estimateTonsFromCuYd(cuyd: number, lbPerCuFt: number) {
  const cuft = cuYdToCuFt(cuyd);
  const lbs = cuft * lbPerCuFt;
  return lbsToTons(lbs);
}

function compute(task: TaskKey, values: Record<string, any>, selectedProduct?: Product): CalcResult {
  // Helper: safe number
  const num = (k: string, fallback = 0) => {
    const v = Number(values[k]);
    return Number.isFinite(v) ? v : fallback;
  };

  // Slots: SOP + manufacturer docs (placeholders now, connect to your real SOP system later)
  const sopLinks: { title: string; url: string }[] = [];
  const manufacturerDocs: { title: string; url: string }[] = [];

  // If product chosen, include its links
  if (selectedProduct) {
    sopLinks.push(...selectedProduct.sopLinks);
    manufacturerDocs.push(...selectedProduct.docs);
  }

  switch (task) {
    case "mulch_soil_gravel": {
      const material = String(values.material || "mulch");
      const areaSqFt = clamp(num("areaSqFt", 0), 0);
      const depthIn = clamp(num("depthIn", 0), 0);
      const wastePct = clamp(num("wastePct", 0), 0);

      const depthFt = inToFt(depthIn);
      const cuft = areaSqFt * depthFt;
      const cuyd = cuFtToCuYd(cuft) * (1 + wastePct / 100);

      const density = DEFAULT_DENSITIES_LB_PER_CUFT[material] ?? 85;
      const estTons = estimateTonsFromCuYd(cuyd, density);

      return {
        title: "Material Quantity",
        lines: [
          { label: "Material", value: material },
          { label: "Cubic Yards (incl. waste)", value: `${roundTo(cuyd, 2)} yd³` },
          { label: "Estimated Tons (rough)", value: `${roundTo(estTons, 2)} tons` },
        ],
        assumptions: [
          "Tons estimate uses a rough density. Use supplier specs for exact conversion.",
          "Waste/overage is optional and depends on job conditions.",
        ],
        safety: [],
        sopLinks: sopLinks.length ? sopLinks : [{ title: "SOP: Material handling & delivery checks (add link)", url: "#" }],
        manufacturerDocs,
      };
    }

    case "paver_base": {
      const areaSqFt = clamp(num("areaSqFt", 0), 0);
      const depthIn = clamp(num("depthIn", 0), 0);
      const compactionPct = clamp(num("compactionPct", 0), 0);
      const wastePct = clamp(num("wastePct", 0), 0);

      const depthFt = inToFt(depthIn);
      const cuft = areaSqFt * depthFt;
      let cuyd = cuFtToCuYd(cuft);
      cuyd = cuyd * (1 + compactionPct / 100) * (1 + wastePct / 100);

      const density = DEFAULT_DENSITIES_LB_PER_CUFT["base_aggregate"] ?? 100;

        assumptions: [

          "Tons estimate uses rough density. Confirm with supplier ticket.",
        ],
        safety: [],
        sopLinks: sopLinks.length ? sopLinks : [{ title: "SOP: Paver base installation checklist (add link)", url: "#" }],
        manufacturerDocs,
      };
    }

    case "retaining_wall_base": {
      const wallLengthFt = clamp(num("wallLengthFt", 0), 0);
      const trenchWidthIn = clamp(num("trenchWidthIn", 0), 0);
      const baseDepthIn = clamp(num("baseDepthIn", 0), 0);
      const wastePct = clamp(num("wastePct", 0), 0);

      const trenchWidthFt = inToFt(trenchWidthIn);
      const baseDepthFt = inToFt(baseDepthIn);

      const cuft = wallLengthFt * trenchWidthFt * baseDepthFt;
      const cuyd = cuFtToCuYd(cuft) * (1 + wastePct / 100);

      const density = DEFAULT_DENSITIES_LB_PER_CUFT["base_aggregate"] ?? 100;
      const estTons = estimateTonsFromCuYd(cuyd, density);

      return {
        title: "Retaining Wall Base Course",
        lines: [

          "Tons estimate is rough—confirm with supplier.",

      // MVP model: adjust coverage based on joint width vs baseline 0.25"

        ],

    case "unit_converter":

      };
    }
  }
}

/** convertAny: MVP unit converter */
function convertAny(value: number, from: string, to: string) {
  if (from === to) return value;

  // length
  const lengthUnits = new Set(["in", "ft", "yd", "cm", "m"]);
  if (lengthUnits.has(from) && lengthUnits.has(to)) {
    // convert to inches
    let inches = value;
    if (from === "ft") inches = ftToIn(value);
    if (from === "yd") inches = ftToIn(value * 3);
    if (from === "cm") inches = cmToIn(value);
    if (from === "m") inches = ftToIn(mToFt(value));

    // inches to target
    if (to === "in") return inches;
    if (to === "ft") return inToFt(inches);
    if (to === "yd") return inToFt(inches) / 3;
    if (to === "cm") return inToCm(inches);
    if (to === "m") return ftToM(inToFt(inches));
  }

  // area
  const areaUnits = new Set(["sqft", "acre"]);
  if (areaUnits.has(from) && areaUnits.has(to)) {
    if (from === "sqft" && to === "acre") return sqFtToAcres(value);
    if (from === "acre" && to === "sqft") return acresToSqFt(value);
  }

  // volume
  const volUnits = new Set(["cuft", "cuyd"]);
  if (volUnits.has(from) && volUnits.has(to)) {
    if (from === "cuft" && to === "cuyd") return cuFtToCuYd(value);
    if (from === "cuyd" && to === "cuft") return cuYdToCuFt(value);
  }

  // liquid
  const liquidUnits = new Set(["gal", "l"]);
  if (liquidUnits.has(from) && liquidUnits.has(to)) {
    if (from === "gal" && to === "l") return gallonsToLiters(value);
    if (from === "l" && to === "gal") return litersToGallons(value);
  }

  // weight
  const weightUnits = new Set(["lb", "ton", "kg"]);
  if (weightUnits.has(from) && weightUnits.has(to)) {
    // convert to lbs
    let lbs = value;
    if (from === "ton") lbs = tonsToLbs(value);
    if (from === "kg") lbs = kgToLbs(value);

    // lbs to target
    if (to === "lb") return lbs;
    if (to === "ton") return lbsToTons(lbs);
    if (to === "kg") return lbsToKg(lbs);
  }

  // fallback: unknown combo
  return NaN;
}

                  key={p.id}
                  onClick={() => applyPreset(p.id)}
                  className={`text-left rounded-xl border p-4 transition-all
                    ${state.presetId === p.id ? "border-emerald-500 bg-emerald-500/10" : "border-neutral-800 hover:border-neutral-700 hover:bg-neutral-900"}
                  `}
                >
                  <div className="flex items-center justify-between">
                    <div className="font-semibold">{p.name}</div>
                    {state.presetId === p.id && <CheckCircle2 className="text-emerald-400" size={18} />}
                  </div>
                  {p.description && <div className="text-sm text-neutral-400 mt-1">{p.description}</div>}
                </button>
              ))}
            </div>
          </Panel>
        )}

        {WIZARD_PAGES[pageIndex] === "Inputs" && (
          <Panel
            title="Enter your measurements"
            desc="Only the inputs that matter for this tool are shown."
          >
            <div className="grid md:grid-cols-2 gap-4">
              <Card className="bg-neutral-950 border-neutral-800">
                <CardHeader className="pb-2">
                  <CardTitle className="text-base">Inputs</CardTitle>
                  <CardDescription className="text-neutral-400">Edit values anytime.</CardDescription>
                </CardHeader>
                <CardContent className="space-y-4">
                  {fields.map(f => {
                    if (f.type === "select") {
                      const value = String(state.values[f.key] ?? "");
                      return (
                        <div key={f.key} className="space-y-1">
                          <div className="text-sm font-medium">{f.label}</div>
                          {f.help && <div className="text-xs text-neutral-500">{f.help}</div>}
                          <Select
                            value={value}
                            onValueChange={(v) => setState(s => ({ ...s, values: { ...s.values, [f.key]: v } }))}
                          >
                            <SelectTrigger className="bg-black border-neutral-800 text-neutral-100">
                              <SelectValue placeholder={`Select ${f.label.toLowerCase()}`} />
                            </SelectTrigger>
                            <SelectContent className="bg-neutral-950 border-neutral-800 text-neutral-100">
                              {f.options.map(o => (
                                <SelectItem key={o.value} value={o.value}>
                                  {o.label}
                                </SelectItem>
                              ))}
                            </SelectContent>
                          </Select>
                        </div>
                      );
                    }

                    const raw = state.values[f.key];
                    const value = raw === undefined ? "" : String(raw);
                    return (
                      <div key={f.key} className="space-y-1">
                        <div className="flex items-center justify-between">
                          <div className="text-sm font-medium">{f.label}</div>
                          {f.unit && <Badge variant="outline" className="border-neutral-700 text-neutral-300">{f.unit}</Badge>}
                        </div>
                        {f.help && <div className="text-xs text-neutral-500">{f.help}</div>}
                        <Input
                          className="bg-black border-neutral-800 text-neutral-100"
                          type="number"
                          value={value}
                          min={f.min}
                          step={f.step ?? 1}
                          onChange={(e) => setState(s => ({ ...s, values: { ...s.values, [f.key]: e.target.value } }))}
                        />
                      </div>
                    );
                  })}
                </CardContent>
              </Card>

              <Card className="bg-neutral-950 border-neutral-800">
                <CardHeader className="pb-2">
                  <CardTitle className="text-base">Quick Notes</CardTitle>
                  <CardDescription className="text-neutral-400">Built-in guidance.</CardDescription>
                </CardHeader>
                <CardContent className="space-y-3 text-sm text-neutral-300">
                  {task === "retaining_wall_base" && (
                    <>
                      <div className="text-neutral-200 font-semibold">Retaining Wall Rule</div>
                      <div className="text-neutral-400">
                        Walls are <span className="text-emerald-300">length-based</span>. That’s why this tool uses wall length × trench width × base depth.
                      </div>
                      <div className="text-neutral-500 text-xs">
                        For “guaranteed” accuracy, store manufacturer-specific trench width and base depth standards per block system.
                      </div>
                    </>
                  )}
                  {intent === "apply_chemical" && (
                    <>
                      <div className="text-neutral-200 font-semibold">Chemical Rule</div>
                      <div className="text-neutral-400">
                        This tool does math. Your <span className="text-emerald-300">label</span> controls legality + safety.
                      </div>
                      <div className="text-neutral-500 text-xs">
                        Later, connect manufacturer label + SDS links and lock rates to verified data.
                      </div>
                    </>
                  )}
                </CardContent>
              </Card>
            </div>
          </Panel>
        )}

        {WIZARD_PAGES[pageIndex] === "Results" && (
          <Panel
            title="Results"
            desc="Copy, verify, and attach to SOPs/jobs (hooks included)."
          >
            {!result ? (
              <div className="text-neutral-400">No result yet.</div>
            ) : (
              <div className="grid md:grid-cols-2 gap-4">
                <Card className="bg-neutral-950 border-neutral-800">
                  <CardHeader className="pb-2">
                    <CardTitle className="text-base">{result.title}</CardTitle>
                    <CardDescription className="text-neutral-400">Your outputs are shown below.</CardDescription>
                  </CardHeader>
                  <CardContent className="space-y-3">
                    {result.lines.map((l, idx) => (
                      <div key={idx} className="flex items-center justify-between rounded-lg border border-neutral-800 bg-black p-3">
                        <div className="text-neutral-400 text-sm">{l.label}</div>
                        <div className="text-neutral-100 font-semibold">{l.value}</div>
                      </div>
                    ))}

                    <div className="pt-2">
                      <Button className={`${neonButtonClass()} w-full`}>
                        Copy Results (hook)
                      </Button>
                      <div className="text-xs text-neutral-500 mt-2">
                        Hook this button into your CompanyHQ clipboard + job attachment system.
                      </div>
                    </div>
                  </CardContent>
                </Card>

                <Card className="bg-neutral-950 border-neutral-800">
                  <CardHeader className="pb-2">
                    <CardTitle className="text-base">References & Safety</CardTitle>
                    <CardDescription className="text-neutral-400">SOP + Manufacturer channels reserved.</CardDescription>
                  </CardHeader>
                  <CardContent className="space-y-4">
                    <Tabs defaultValue="assumptions">
                      <TabsList className="bg-black border border-neutral-800">
                        <TabsTrigger value="assumptions">Assumptions</TabsTrigger>
                        <TabsTrigger value="safety">Safety</TabsTrigger>
                        <TabsTrigger value="sops">SOPs</TabsTrigger>
                        <TabsTrigger value="docs">Docs</TabsTrigger>
                      </TabsList>

                      <TabsContent value="assumptions" className="mt-3 space-y-2">
                        {result.assumptions.length ? result.assumptions.map((a, i) => (
                          <div key={i} className="text-sm text-neutral-300 rounded-lg border border-neutral-800 bg-black p-3">
                            {a}
                          </div>
                        )) : <div className="text-sm text-neutral-500">No assumptions.</div>}
                      </TabsContent>

                      <TabsContent value="safety" className="mt-3 space-y-2">
                        {result.safety.length ? result.safety.map((s, i) => (
                          <div key={i} className="text-sm text-neutral-300 rounded-lg border border-neutral-800 bg-black p-3">
                            {s}
                          </div>
                        )) : <div className="text-sm text-neutral-500">No safety notes for this tool.</div>}
                      </TabsContent>

                      <TabsContent value="sops" className="mt-3 space-y-2">
                        {result.sopLinks.map((x, i) => (
                          <div key={i} className="flex items-center justify-between rounded-lg border border-neutral-800 bg-black p-3">
                            <div className="flex items-center gap-2">
                              <FileText size={16} className="text-emerald-400" />
                              <span className="text-sm text-neutral-200">{x.title}</span>
                            </div>
                            <Button variant="outline" className={neonOutlineButtonClass()} size="sm">
                              <LinkIcon size={14} className="mr-1" /> Open
                            </Button>
                          </div>
                        ))}
                        <div className="text-xs text-neutral-500">
                          Replace “Open” with navigation to your SOP detail page.
                        </div>
                      </TabsContent>

                      <TabsContent value="docs" className="mt-3 space-y-2">
                        {result.manufacturerDocs.length ? result.manufacturerDocs.map((x, i) => (
                          <div key={i} className="flex items-center justify-between rounded-lg border border-neutral-800 bg-black p-3">
                            <div className="flex items-center gap-2">
                              <FileText size={16} className="text-emerald-400" />
                              <span className="text-sm text-neutral-200">{x.title}</span>
                            </div>
                            <Button variant="outline" className={neonOutlineButtonClass()} size="sm">
                              <LinkIcon size={14} className="mr-1" /> View
                            </Button>
                          </div>
                        )) : (
                          <div className="text-sm text-neutral-500">
                            No docs linked yet. (Add SDS/Label URLs when you build the data library.)
                          </div>
                        )}

                        <Separator className="bg-neutral-800 my-2" />

                        <div className="rounded-lg border border-neutral-800 bg-black p-3">
                          <div className="text-sm font-semibold text-neutral-200">Manual Update (for now)</div>
                          <div className="text-xs text-neutral-500 mt-1">
                            Until your auto-updating agent is built, allow users to flag/confirm data.
                          </div>
                          <div className="mt-3 flex gap-2">
                            <Button className={neonButtonClass()} size="sm">Check for Updates (hook)</Button>
                            <Button variant="outline" className={neonOutlineButtonClass()} size="sm">Flag Outdated (hook)</Button>
                          </div>
                        </div>
                      </TabsContent>
                    </Tabs>
                  </CardContent>
                </Card>
              </div>
            )}
          </Panel>
        )}

        {/* Footer controls */}
        <div className="flex items-center justify-between">
          <Button
            variant="outline"
            className={neonOutlineButtonClass()}
            onClick={goBack}
            disabled={pageIndex === 0}
          >
            <ArrowLeft className="mr-2" size={16} /> Back
          </Button>

          <div className="flex gap-2">
            {WIZARD_PAGES[pageIndex] !== "Results" ? (
              <Button
                className={neonButtonClass()}
                onClick={goNext}
                disabled={!canNext}
              >
                Next <ArrowRight className="ml-2" size={16} />
              </Button>
            ) : (
              <Button
                className={neonButtonClass()}
                onClick={() => {
                  // restart keeping selections (optional)
                  setPageIndex(0);
                  setState({ values: {} });
                }}
              >
                Start New Calculation
              </Button>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
